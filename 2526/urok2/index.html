            int a = st.top(); st.pop();
            if (s == "+") {
                st.push(a + b);
            } else if (s == "*") {
                st.push(a * b);
            } else if (s == "/") {
                st.push(a / b);
            } else if (s == "-") {
                st.push(a - b);
            }
        } else {
            int n = stoi(s);
            st.push(n);
        }
    }

    cout << st.top() << endl;

    return 0;
}
